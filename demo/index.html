Bootstrap-Strength.js

jQuery plugin to calculate string or password strength with Bootstrap progress bar full customizable.

Thanks to @aarondo project that inspire a Bootstrap dedicated plugin.

Default requires for validation and option variables

  • At least 8 characters
  • 1 Uppercase letter
  • 1 lowercase letter
  • 1 number
  • 1 special character such as [!,%,&,@,#,$,^,*,?,_,~]
$.bootstrapStrength.defaults = {
	stripped: true,
	active: true,
	slimBar: true,
	minLenght: 8,
	upperCase: 1,
	upperReg: "[A-Z]",
	lowerCase: 1,
	lowerReg: "[a-z]",
	numbers: 1,
	numberReg: "[0-9]",
	specialchars: 1,
	specialReg: "[!,%,&,@,#,$,^,*,?,_,~]",
	topMargin: "5px;",
	meterClasses: {
		weak: "progress-bar-danger",
		medium: "progress-bar-warning",
		good: "progress-bar-success"
	}
};

Basic usage

$('.passMeter').bootstrapStrength();

Using standard progress bar

$('.passMeterSlim').bootstrapStrength({
	slimBar: false
});
